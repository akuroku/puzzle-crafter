2   package ohha.puzzlecrafter.grid;
3   
4   import java.util.Set;
5   import java.util.HashSet;
6   
7   
8   public class Partition {
9       
10      private Set<Region> regions;
11      
12      
13      public Partition() {
14          regions = new HashSet<>();
15      }
16      
17      
18      public void addRegion(Region r) {
19          regions.add(r);
20      }
21      
22      public Set<Region> getRegions() {
23          return regions;
24      }
25      
26      
27      public boolean contains(Region r) {
28          return regions.contains(r);
29      }
30      
31      
32      public void createRectangularPartition(int regionHeight, int regionWidth, int partitionHeight, int partitionWidth) {
33          regions = new HashSet<>();
34          
35          for (int y = 0; y < partitionHeight; y++) {
36              for (int x = 0; x < partitionWidth; x++) {
37                  addRegion(createRectangularRegion(regionHeight, regionWidth, x * regionWidth, y * regionHeight));
38              }
39          }
40      }
41      
42      private Region createRectangularRegion(int regionHeight, int regionWidth, int topLeftX, int topLeftY) {
43          Region region = new Region();
44          
45          for (int y = 0; y < regionHeight; y++) {
46              for (int x = 0; x < regionWidth; x++) {
47                  region.addCell(new Cell(topLeftX + x, topLeftY + y));
48              }
49          }
50          return region;
51      }
52      
53      
54      public Region getRegionOf(Cell c) {
55          for (Region region : regions) {
56              if (region.contains(c)) {
57                  return region;
58              }
59          }
60          return null; // this should never be returned with a valid partition
61      }
62      
63      
64      @Override
65      public String toString() {
66          String s = "";
67          for (Region region : regions) {
68              s += region + "\n";
69          }
70          return s;
71      }
72  }
