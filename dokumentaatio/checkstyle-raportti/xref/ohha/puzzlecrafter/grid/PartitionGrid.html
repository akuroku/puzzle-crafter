5   public class PartitionGrid extends Grid {
6       
7       private Partition partition;
8       
9       
10      // initialise grid with given regions
11      public PartitionGrid(int height, int width, Partition partition) {
12          super(height, width);
13          this.partition = partition;
14      }
15      
16      // initialise grid with rectangular regions with dimensions height and width
17      public PartitionGrid(int gridHeight, int gridWidth, int regionHeight, int regionWidth) {
18          super(gridHeight, gridWidth);
19          this.partition = new Partition();
20          partition.createRectangularPartition(regionHeight, regionWidth, getHeight() / regionHeight, getWidth() / regionWidth);
21      }
22      
23      public Partition getPartition() {
24          return partition;
25      }
26      
27      
28      public Region getRegionOf(Cell c) {
29          return partition.getRegionOf(c);
30      }
31      
32      
33      public boolean isDuplicatedOnRegion(Cell c) {
34          if (!isFilledIn(c)) {
35              return false;
36          }
37          
38          for (Cell cell : getRegionOf(c).getCells()) {
39              if (cell.equals(c)) {
40                  continue;
41              }
42              if (getValueAt(c) == getValueAt(cell)) {
43                  return true;
44              }
45          }
46          return false;
47      }
48  }
