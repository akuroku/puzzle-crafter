35  /**
36   * Acts as the main window of the program, from which individual puzzle styles
37   * are created.
38   * The layout design is completed, but as the puzzle editor itself is
39   * unfinished, the puzzle creation functionality is yet lacking.
40   * <p>
41   * Not yet finished.
42   */
43  public class MainWindow implements Runnable {
44      
45      private JFrame frame;
46      
47      private JComboBox puzzleDropDown;
48      private JSpinner gridHeightSpinner;
49      private JSpinner gridWidthSpinner;
50      private JSpinner cellSizeSpinner;
51      
52      
53      @Override
54      public void run() {
55          frame = new JFrame("Puzzlecrafter");                // set name displayed on window
56          // frame.setPreferredSize(new Dimension(400, 300)); // set dimensions of window
57          frame.setResizable(false);                          // make window unresizable
58          
59          frame.setDefaultCloseOperation(WindowConstants.EXIT_ON_CLOSE);
60          
61          createComponents(frame.getContentPane());
62          
63          frame.pack();
64          frame.setVisible(true);
65      }
66      
67      
68      private void createComponents(Container container) {
69          
70          UIManager.put("Spinner.editorAlignment", JTextField.LEFT);  // set spinner values on the left
71          
72          
73          // Create puzzle selection bar on top
74          
75          puzzleDropDown = new JComboBox(Puzzle.STYLES);
76          puzzleDropDown.setSelectedIndex(0);
77          
78          JLabel selectPuzzleLabel = new JLabel("Select puzzle style:", JLabel.CENTER);
79          selectPuzzleLabel.setLabelFor(puzzleDropDown);
80          
81          JPanel puzzleSelectionPane = new JPanel(new GridLayout(1, 2, 50, 0));   // rows, columns, horizontal gap, vertical gap
82          
83          puzzleSelectionPane.setBorder(BorderFactory.createEmptyBorder(10, 20, 10, 20)); // top, left, bottom, right
84  
85          puzzleSelectionPane.add(selectPuzzleLabel);     // text
86          puzzleSelectionPane.add(puzzleDropDown);        // dropdown menu
87          
88          
89          
90          // Create puzzle creation button on bottom
91          
92          JButton createPuzzleButton = new JButton("Create");
93          
94          JPanel puzzleCreationPane = new JPanel(new FlowLayout());
95          
96          puzzleCreationPane.setBorder(BorderFactory.createEmptyBorder(7, 20, 10, 20)); // top, left, bottom, right
97          
98          puzzleCreationPane.add(createPuzzleButton);
99          
100         
101 
102         // Create puzzle settings pane
103         // todo: make one grid dimension spinner inactive and copy the other when puzzle style requires square grid
104         
105         JPanel puzzleSettingsPane = new JPanel(new GridLayout(3, 2, 5, 5));
106         
107         puzzleSettingsPane.setBorder(BorderFactory.createCompoundBorder(
108                 BorderFactory.createEmptyBorder(0, 10, 0, 10),              // outer empty buffer // top, left, bottom, right
109                 BorderFactory.createCompoundBorder(
110                         BorderFactory.createLineBorder(Color.LIGHT_GRAY),   // outline
111                         BorderFactory.createEmptyBorder(5, 5, 5, 5)         // inner empty buffer
112         )));
113         
114         gridHeightSpinner = new JSpinner(new SpinnerNumberModel(Grid.DEFAULT_SIZE, Grid.MIN_SIZE, Grid.MAX_SIZE, 1));   // default, min, max, step
115         JLabel gridHeightLabel = new JLabel("Grid height:", JLabel.RIGHT);
116         gridHeightLabel.setToolTipText("Input an integer between " + Grid.MIN_SIZE + " and " + Grid.MAX_SIZE + " to specify the height of the grid in cells.");
117         
118         gridWidthSpinner = new JSpinner(new SpinnerNumberModel(Grid.DEFAULT_SIZE, Grid.MIN_SIZE, Grid.MAX_SIZE, 1));   // default, min, max, step
119         JLabel gridWidthLabel = new JLabel("Grid width:", JLabel.RIGHT);
120         gridWidthLabel.setToolTipText("Input an integer between " + Grid.MIN_SIZE + " and " + Grid.MAX_SIZE + " to specify the width of the grid in cells.");
121         
122         cellSizeSpinner = new JSpinner(new SpinnerNumberModel(Drawer.DEFAULT_CELL_SIZE, Drawer.MIN_CELL_SIZE, Drawer.MAX_CELL_SIZE, 1));   // default, min, max, step
123         JLabel cellSizeLabel = new JLabel("Cell size:", JLabel.RIGHT);
124         cellSizeLabel.setToolTipText("Input an integer between " + Drawer.MIN_CELL_SIZE + " and " + Drawer.MAX_CELL_SIZE + " to specify the height and width of a cell in pixels.");
125         
126         puzzleSettingsPane.add(gridHeightLabel);
127         puzzleSettingsPane.add(gridHeightSpinner);
128         
129         puzzleSettingsPane.add(gridWidthLabel);
130         puzzleSettingsPane.add(gridWidthSpinner);
131         
132         puzzleSettingsPane.add(cellSizeLabel);
133         puzzleSettingsPane.add(cellSizeSpinner);
134         
135         
136         
137         // Add panes to main window
138         
139         container.add(puzzleSelectionPane, BorderLayout.NORTH);
140         container.add(puzzleSettingsPane, BorderLayout.CENTER);
141         container.add(puzzleCreationPane, BorderLayout.SOUTH);
142         
143         
144         
145         // Create event listeners
146         
147         puzzleDropDown.addItemListener((ItemEvent ie) -> {
148             switch ((String) ie.getItem()) {
149                 case "Sudoku": {
150                     gridHeightLabel.setText("Grid size:");
151                     gridWidthLabel.setText("Region width:");
152                     break;
153                 }
154                 case "Fillomino": {
155                     gridHeightLabel.setText("Grid height:");
156                     gridWidthLabel.setText("Grid width:");
157                     break;
158                 }
159                 default:
160                     break;
161             }
162         });
163         
164         createPuzzleButton.addActionListener((ActionEvent ae) -> {
165             switch((String) puzzleDropDown.getSelectedItem()) {
166                 case "Fillomino": {
167                     makeFillomino();
168                     break;
169                 }
170                 case "Sudoku": {
171                     makeSudoku();
172                     break;
173                 }
174                 default: {
175                     JOptionPane.showMessageDialog(frame, "there is an error?", "error", JOptionPane.ERROR_MESSAGE);
176                 }
177             }
178         });
179     }
180     
181     private void makeFillomino() {
182         SwingUtilities.invokeLater(new PuzzleEditorWindow(
183                 new FillominoDrawer(
184                         new Fillomino((int) gridHeightSpinner.getValue(), (int) gridWidthSpinner.getValue()), 
185                         (int) cellSizeSpinner.getValue()
186                 )
187         ));
188     }
189     
190     private void makeSudoku() {
191         int gridSize = (int) gridHeightSpinner.getValue();
192         int regionWidth = (int) gridWidthSpinner.getValue();
193         int regionHeight = gridSize / regionWidth;
194         
195         if (gridSize != regionHeight * regionWidth) {
196             JOptionPane.showMessageDialog(frame, "Region width doesn't divide grid size evenly!");
197             return;
198         }
199         
200         SwingUtilities.invokeLater(new PuzzleEditorWindow(
201                 new SudokuDrawer(
202                         new Sudoku(gridSize, regionHeight, regionWidth), 
203                         (int) cellSizeSpinner.getValue()
204                 )
205         ));
206     }
207     
208     public JFrame getFrame() {
209         return frame;
210     }
211 }
