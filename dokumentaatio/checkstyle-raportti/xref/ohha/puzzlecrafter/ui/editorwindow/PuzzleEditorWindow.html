3   
4   import java.awt.BorderLayout;
5   import java.awt.Container;
6   import java.awt.event.ActionEvent;
7   import java.awt.event.KeyEvent;
8   import java.awt.event.MouseWheelEvent;
9   import java.util.List;
10  import javax.swing.JFrame;
11  import javax.swing.JMenu;
12  import javax.swing.JMenuBar;
13  import javax.swing.JMenuItem;
14  import javax.swing.JOptionPane;
15  import javax.swing.JTabbedPane;
16  import ohha.puzzlecrafter.Solver;
17  import ohha.puzzlecrafter.puzzles.Puzzle;
18  
19  import ohha.puzzlecrafter.ui.drawers.Drawer;
20  
21  
22  /**
23   * Acts as the window where puzzles are created and solved in.
24   * Not yet finished.
25   */
26  public class PuzzleEditorWindow implements Runnable {
27      
28      private JFrame frame;
29      private JTabbedPane tabbedPane;
30      private JMenuBar menuBar;
31      
32      private Drawer drawer;
33      
34      
35      public PuzzleEditorWindow(Drawer drawer) {
36          this.drawer = drawer;
37      }
38      
39      
40      private GridPane getActiveGridPane() {
41          return (GridPane) tabbedPane.getSelectedComponent();
42      }
43      
44      
45      @Override
46      public void run() {
47          frame = new JFrame(drawer.getPuzzle().getName());   // set name displayed on window
48          // frame.setPreferredSize(new Dimension(400, 300)); // set dimensions of window
49          frame.setResizable(false);                          // make window unresizable
50          
51          createComponents(frame.getContentPane());
52          
53          drawer = null;  // don't keep the drawer references without need
54          
55          /* a terrible hack
56          a mouse wheel listener attached on gridPane will remain dormant as I
57          can't get the panel to get focus so I'm attaching the mouse wheel
58          listener here and passing the event on
59          */
60          frame.addMouseWheelListener((MouseWheelEvent mwe) -> {
61              mwe.consume();
62              ((GridPane) tabbedPane.getSelectedComponent()).getEditor().mouseWheelMoved2(mwe);
63          });
64          
65          frame.pack();
66          frame.setVisible(true);
67      }
68      
69      
70      private void createComponents(Container container) {
71          container.setLayout(new BorderLayout());
72          
73          
74          // create tabbed pane
75          tabbedPane = new JTabbedPane();
76          tabbedPane.setTabPlacement(JTabbedPane.TOP);
77          tabbedPane.setTabLayoutPolicy(JTabbedPane.SCROLL_TAB_LAYOUT);
78          
79          tabbedPane.add("1", new GridPane(drawer, "1"));
80          
81          container.add(tabbedPane);
82          
83          
84          // create menu bar
85          menuBar = new JMenuBar();
86          
87          
88          // create tab menu
89          JMenu tabMenu = new JMenu("Tab");
90          tabMenu.setMnemonic(KeyEvent.VK_T);
91          
92          
93          JMenuItem closeButton = new JMenuItem("Close");
94          closeButton.setMnemonic(KeyEvent.VK_C);
95          closeButton.addActionListener((ActionEvent ae) -> {
96              tabbedPane.removeTabAt(tabbedPane.getSelectedIndex());
97              
98              if (tabbedPane.getTabCount() == 0) {
99                  frame.dispose();
100             }
101         });
102         tabMenu.add(closeButton);
103         
104         
105         JMenuItem renameButton = new JMenuItem("Rename");
106         renameButton.setMnemonic(KeyEvent.VK_R);
107         renameButton.addActionListener((ActionEvent ae) -> {
108             String code = (String) JOptionPane.showInputDialog(
109                     frame,
110                     "Enter a name for the tab:",
111                     "Rename",
112                     JOptionPane.PLAIN_MESSAGE);
113             
114             if (code == null || code.equals("")) {
115                 return;
116             }
117             
118             tabbedPane.setTitleAt(tabbedPane.getSelectedIndex(), code);
119             getActiveGridPane().setCode(code);
120         });
121         tabMenu.add(renameButton);
122         
123         
124         JMenuItem copyButton = new JMenuItem("Copy");
125         copyButton.addActionListener((ActionEvent ae) -> {
126             copy();
127         });
128         tabMenu.add(copyButton);
129         
130         menuBar.add(tabMenu);
131         
132         
133         // create solve menu
134         
135         JMenu solveMenu = new JMenu("Solve");
136         tabMenu.setMnemonic(KeyEvent.VK_S);
137         
138         
139         JMenuItem countSolutionsButton = new JMenuItem("Count solutions");
140         countSolutionsButton.setMnemonic(KeyEvent.VK_C);
141         countSolutionsButton.addActionListener((ActionEvent ae) -> {
142             int solutions = new Solver(getActiveGridPane().getDrawer().getPuzzle()).getAmountOfSolutions();
143             
144             String message;
145             
146             if (solutions == 0) {
147                 message = "No solutions found!";
148             } else if (solutions == 1) {
149                 message = "One solution found!";
150             } else {
151                 message = solutions + " solutions found!";
152             }
153             JOptionPane.showMessageDialog(frame, message);
154         });
155         solveMenu.add(countSolutionsButton);
156         
157         
158         JMenuItem getSolutionsButton = new JMenuItem("Generate solutions");
159         getSolutionsButton.setMnemonic(KeyEvent.VK_S);
160         getSolutionsButton.addActionListener((ActionEvent ae) -> {
161             List<Puzzle> solutions = new Solver(getActiveGridPane().getDrawer().getPuzzle()).getSolutions();
162             
163             if (solutions.isEmpty()) {
164                 JOptionPane.showMessageDialog(frame, "No solutions found!");
165                 return;
166             }
167             
168             String code;
169             int i = 1;
170             
171             for (Puzzle puzzle : new Solver(getActiveGridPane().getDrawer().getPuzzle().deepCopy()).getSolutions()) {
172                 code = getActiveGridPane().getCode() + ".s." + i++;
173                 tabbedPane.add(code, new GridPane(getActiveGridPane().getDrawer().duplicate(puzzle), code));
174             }
175         });
176         solveMenu.add(getSolutionsButton);
177         
178         menuBar.add(solveMenu);
179         
180         
181         frame.setJMenuBar(menuBar);
182     }
183     
184     
185     private void copy() {
186         GridPane copy = getActiveGridPane().makeCopy();
187         tabbedPane.add(copy.getCode(), copy);
188     }
189 }
