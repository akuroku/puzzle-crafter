4   import java.awt.BorderLayout;
5   import java.awt.Container;
6   import java.awt.event.ActionEvent;
7   import java.awt.event.KeyEvent;
8   import java.awt.event.MouseWheelEvent;
9   import javax.swing.JFrame;
10  import javax.swing.JMenu;
11  import javax.swing.JMenuBar;
12  import javax.swing.JMenuItem;
13  import javax.swing.JOptionPane;
14  import javax.swing.JTabbedPane;
15  
16  import ohha.puzzlecrafter.ui.drawers.Drawer;
17  
18  
19  /**
20   * Acts as the window where puzzles are created and solved in.
21   * Not yet finished.
22   */
23  public class PuzzleEditorWindow implements Runnable {
24      
25      private JFrame frame;
26      private JTabbedPane tabbedPane;
27      private JMenuBar menuBar;
28      
29      private Drawer drawer;
30      
31      
32      public PuzzleEditorWindow(Drawer drawer) {
33          this.drawer = drawer;
34      }
35      
36      
37      private GridPane getActiveGridPane() {
38          return (GridPane) tabbedPane.getSelectedComponent();
39      }
40      
41      
42      @Override
43      public void run() {
44          frame = new JFrame(drawer.getPuzzle().getName());   // set name displayed on window
45          // frame.setPreferredSize(new Dimension(400, 300)); // set dimensions of window
46          frame.setResizable(false);                          // make window unresizable
47          
48          createComponents(frame.getContentPane());
49          
50          drawer = null;  // don't keep the drawer references without need
51          
52          /* a terrible hack
53          a mouse wheel listener attached on gridPane will remain dormant as I
54          can't get the panel to get focus so I'm attaching the mouse wheel
55          listener here and passing the event on
56          */
57          frame.addMouseWheelListener((MouseWheelEvent mwe) -> {
58              mwe.consume();
59              ((GridPane) tabbedPane.getSelectedComponent()).getEditor().mouseWheelMoved2(mwe);
60          });
61          
62          frame.pack();
63          frame.setVisible(true);
64      }
65      
66      
67      private void createComponents(Container container) {
68          container.setLayout(new BorderLayout());
69          
70          
71          // create tabbed pane
72          tabbedPane = new JTabbedPane();
73          tabbedPane.setTabPlacement(JTabbedPane.TOP);
74          tabbedPane.setTabLayoutPolicy(JTabbedPane.SCROLL_TAB_LAYOUT);
75          
76          tabbedPane.add("1", new GridPane(drawer, "1"));
77          
78          container.add(tabbedPane);
79          
80          
81          // create menu bar
82          menuBar = new JMenuBar();
83          
84          
85          JMenu copyMenu = new JMenu("Tab");
86          copyMenu.setMnemonic(KeyEvent.VK_T);
87          
88          
89          JMenuItem closeButton = new JMenuItem("Close");
90          closeButton.setMnemonic(KeyEvent.VK_C);
91          closeButton.addActionListener((ActionEvent ae) -> {
92              tabbedPane.removeTabAt(tabbedPane.getSelectedIndex());
93              
94              if (tabbedPane.getTabCount() == 0) {
95                  frame.dispose();
96              }
97          });
98          copyMenu.add(closeButton);
99          
100         
101         JMenuItem renameButton = new JMenuItem("Rename");
102         renameButton.setMnemonic(KeyEvent.VK_R);
103         renameButton.addActionListener((ActionEvent ae) -> {
104             String code = (String) JOptionPane.showInputDialog(
105                     frame,
106                     "Enter a name for the tab:",
107                     "Rename",
108                     JOptionPane.PLAIN_MESSAGE);
109             
110             if (code == null || code.equals("")) {
111                 return;
112             }
113             
114             tabbedPane.setTitleAt(tabbedPane.getSelectedIndex(), code);
115             getActiveGridPane().setCode(code);
116         });
117         copyMenu.add(renameButton);
118         
119         
120         JMenuItem copyButton = new JMenuItem("Copy");
121         copyButton.addActionListener((ActionEvent ae) -> {
122             copy();
123         });
124         copyMenu.add(copyButton);
125         
126         menuBar.add(copyMenu);
127         
128         container.add(menuBar, BorderLayout.NORTH);
129     }
130     
131     
132     private void copy() {
133         GridPane copy = getActiveGridPane().makeCopy();
134         tabbedPane.add(copy.getCode(), copy);
135     }
136 }
